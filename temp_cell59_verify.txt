from linearmodels.panel import compare
from analysis.regression_utils import prepare_regression_data, run_panel_ols, generate_marginal_effects, LATEX_LABEL_MAP
from clean.utils import load_config

config = load_config()
# Restrict to primary indices as per user preference
indices = ['KOFGI', 'KOFEcGI', 'KOFSoGI', 'KOFPoGI']
interaction_results = {}

for idx_name in indices:
    if idx_name not in master_regimes.columns:
        continue
    # Prepare variables with lags
    current_ctrl_vars = [idx_name] + config.get('controls')
    reg_data = create_lags(master_regimes, current_ctrl_vars, lags=[1])
    
    g_var = f'{idx_name}_lag1'
    lagged_ctrls = [f'{v}_lag1' for v in current_ctrl_vars if v != idx_name]
    
    # Selectively create Conservative and Liberal interactions
    reg_data['int_conservative'] = reg_data[g_var] * reg_data['regime_conservative']
    reg_data['int_liberal'] = reg_data[g_var] * reg_data['regime_liberal']
    
    custom_lagged_ctrls = ['int_conservative', 'int_liberal'] + lagged_ctrls
    
    # Use regression utils with interactions=False (manual interactions included in custom_lagged_ctrls)
    ols_data, exog_vars = prepare_regression_data(reg_data, 'sstran', g_var, custom_lagged_ctrls, interactions=False)
    
    # Map column headers to readable names
    header_name = LATEX_LABEL_MAP.get(g_var, idx_name).replace('_{t-1}', '').replace('$', '')
    interaction_results[header_name] = run_panel_ols(ols_data, 'sstran', exog_vars)
    
    # Marginal effects summary
    print(f'\n{'='*20} {header_name} {'='*20}')
    me_table = generate_marginal_effects(interaction_results[header_name], g_var)
    display(me_table.round(4))

# Summary Comparison
interaction_comparison = compare(interaction_results, stars=True)
display(interaction_comparison)

# Export Summary Table to LaTeX
output_file = '../outputs/tables/interaction_regression_table.tex'
os.makedirs(os.path.dirname(output_file), exist_ok=True)

latex_str = interaction_comparison.summary.as_latex()
for old, new in LATEX_LABEL_MAP.items():
    latex_str = latex_str.replace(old, new)

with open(output_file, 'w', encoding='utf-8') as f:
    f.write(latex_str)
print(f'\n✅ Interaction table saved to: {output_file}')
